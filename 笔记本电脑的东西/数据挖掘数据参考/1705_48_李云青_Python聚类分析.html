#定义函数：元素按照出现次数排序（降序）
def sort_count(list):
    '''
    param list: 元素为int型的列表
    return: 元素按照出现次数排序（降序）
    '''
    list2 = []
    result = []
    for i in set(list):
        list2.append([list.count(i),i])

    list2.sort(reverse=True)
    # print(list2[1][1])
    for count,num in list2:
        result.append(num)
    return result

# 定义函数：获取每个类代表的是什么数字（标签）
def show(label):
    numbers=[]
    re={} #键为数字，值为标签
    for i in range(10):
        index=np.where(y==i) #真实数字i的索引
        values=(label[index]).tolist() #根据索引获取对应标签
        number_label=max(values,key=values.count) #出现次数最多的标签
        
        #判断numbers是否已存在该标签。如果是，则选取频数稍微小一点的标签
        if number_label in numbers:
            sort=[]
            length=[]          
            n=np.where((np.array(numbers))==number_label)
            num=n[0].tolist()
            num.append(i)
            for j in num:
                index_=np.where(y==j) 
                values_=(label[index_]).tolist() #根据索引获取对应标签
                v=sort_count(values_)
                length.append(len(v))
                sort.append(v)              
            # print(sort,num,length)
            
            l=np.where((np.array(length))==max(length))
            w=l[0][0]
            for t in np.arange(1,max(length)):
                # print(t)
                if t<=min(length)-1:
                    if sort[0][t] not in numbers:
                        b=num[0]
                        numbers[b]=sort[0][t]
                        break
                    if sort[1][t] not in numbers:
                        number_label=sort[1][t]
                        break
                else:                   
                    if sort[w][t] not in numbers:
                        number_label=sort[w][t]
                        break
                    
        numbers.append(number_label)
    #字典
    for i,j in zip(range(10),numbers):
        re[i]=j
    return re

#定义标签换成数字
def transfrom(y_pred,M):
    res=np.arange(len(y_pred)).reshape(len(y_pred),)
    for i,j in zip(M.values(),M.keys()):
        index=np.where(y_pred==i)
        res[index]=j
    return res

def error_rate(predict,labels):
    # 计算错误率,针对二分类问题,分类标签为0或1
    # 输入参数格式为numpy的向量
    error =0.0
    for i in range(len(predict)):
        predict[i] = round(predict[i]) 
        if predict[i]!=labels[i]:
            error+=1
    return error/len(predict)
