import pandas as pd
import re
import jieba


def data_process(file='D:\数据挖掘\python编程基础\.idea\附件2.xlsx'):
    data = pd.read_excel(file,index_col=0)
    data=data[['留言详情','一级标签']]
    data.columns = ['content', 'labels']
    #将一级标签转化成id
    data['id']=data['labels'].factorize()[0]
    id_data=data[['labels','id']].drop_duplicates().sort_values('id').reset_index(drop=True)
    to_id=dict(id_data.values)
    id_to=dict(id_data[['id','labels']].values)

    data['labels'].value_counts()  #标签类别数
    # 有的分类留言条数少，有的留言条数多，所以每个分类抽取不同的条数进行建模
    n1 = 1200
    n2=800
    n3=600

    #根据分类进行抽样
    a=data[data['labels']=='城乡建设'].sample(n1)
    b=data[data['labels']=='环境保护'].sample(n2)
    c=data[data['labels']=='交通运输'].sample(n3)
    d=data[data['labels']=='教育文体'].sample(n1)
    e=data[data['labels']=='劳动和社会保障'].sample(n2)
    f=data[data['labels']=='商贸旅游'].sample(n3)
    g=data[data['labels']=='卫生计生'].sample(n2)

    #进行拼接
    data_new=pd.concat([a,b,c,d,e,f,g],axis=0)
    #数据预处理
    data_dup = data_new['content'].drop_duplicates()
    data_qumin = data_dup.apply(lambda x: re.sub(u"[^a-zA-Z0-9\u4E00-\u9FA5]", '', x))

    data_cut = data_qumin.apply(lambda x: jieba.lcut(x))

    stopWords = pd.read_csv('D:\数据挖掘\python编程基础\.idea\stopword.txt', encoding='GB18030', sep='hahaha', header=None,engine='python')
    stopWords = ['：', '。', '！', '？', ' ', '【', '】', '》', '《','年','月','日','0'] + list(stopWords.iloc[:, 0])
    data_after_stop = data_cut.apply(lambda x: [i for i in x if i not in stopWords])
    labels = data_new.loc[data_after_stop.index, 'labels']
    adata = data_after_stop.apply(lambda x: ' '.join(x))

    return adata, data_after_stop, labels

data_process()

# 模型建立与预测
from sklearn.svm import LinearSVC
from data_process import data_process
from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import CountVectorizer, TfidfTransformer

adata, data_after_stop, lables = data_process()
data_tr, data_te, labels_tr, labels_te = train_test_split(adata, lables, test_size=0.2)

countVectorizer = CountVectorizer()
data_tr = countVectorizer.fit_transform(data_tr)
X_tr = TfidfTransformer().fit_transform(data_tr.toarray()).toarray()

data_te = CountVectorizer(vocabulary=countVectorizer.vocabulary_).fit_transform(data_te)
X_te = TfidfTransformer().fit_transform(data_te.toarray()).toarray()

model = LinearSVC()
model.fit(X_tr, labels_tr)
model.score(X_te, labels_te)

print("模型预测值为：",model.score(X_te, labels_te))
